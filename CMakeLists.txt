cmake_minimum_required(VERSION "3.12")

project("cmake-exercise-parajuba" VERSION 1.0)

  HINTS ${DEAL_II_DIR} ../ ../../ $ENV{DEAL_II_DIR}
  )

                filesystem/filesystem.cpp
                flatset/flatset.cpp
